/* 
 * This program is meant to illustrate concepts related to data, types, variables, and expressions along with input and output. 
 */
package Testing_Bed;
import java.util.Scanner;

public class DataPlay {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        System.out.print("Enter an integer: ");
        int num1 = scanner.nextInt();
        System.out.print("Enter a second integer: ");
        int num2 = scanner.nextInt();
        System.out.println("The values of the read integers are: ");
        System.out.println("num1 --> " + num1);
        System.out.println("num2 --> " + num2);

        /* 
         * We will now experiment with computing the values of some expressions using the two integers we entered 
         */

        //Sum of two numbers
        int sum = num1 + num2;
        System.out.println("Sum of the numbers: " + sum);
        //Product of the two numbers
        int product = num1 * num2;
        System.out.println("Product of the numbers: " + product);
        //Integer quotient of the two numbers
        int integerQuotient  = num1 / num2;
        System.out.println("Integer quotient of the numbers: " + integerQuotient);
        //Integer quotient of the two numbers using double
        double doubleQuotient = num1 / num2;
        System.out.println("Integer quotient using a double: " + doubleQuotient);
        //True quotient of the two numbers
        double trueQuotient = (double)num1 / (double)num2;
        System.out.println("True Quotient: " + trueQuotient);
        //The remainder of integer division, the modulus operator
        int mod = num1 % num2;
        System.out.println(num1 + " modulo " + num2 + " = " + mod);
        //The real average of the two numbers
        double average = ((num1 + num2) / 2.0);
        System.out.println("The real average of the two numbers: " + average);
        //The area of a circle whose radius is the difference of the numbers
        double radius = Math.abs(num1 - num2);
        double area = Math.PI * Math.pow(radius, 2);
        System.out.println("The area of a circle with radius: " + radius + " is: " + area);
        //What is this?
        String pair = "(" + num1 + ", " + num2 + ")";
        System.out.println("pair --> " + pair);
    }
}
